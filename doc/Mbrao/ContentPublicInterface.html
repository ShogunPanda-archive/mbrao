def enabled_for_locales?(*locales)
  locales = locales.flatten.ensure_array(nil, false, false, true) {|l| l.ensure_string.strip }.reject {|l| l == "*" }
  @locales.blank? || locales.blank? || (@locales & locales).present?
end
