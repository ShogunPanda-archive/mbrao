def is_email?(text)
  regex = /^([a-z0-9_\.\-\+]+)@([\da-z\.\-]+)\.([a-z\.]{2,6})$/i
  text.ensure_string.strip =~ regex
end

def is_url?(text)
  regex = /
    ^(
      ([a-z0-9\-]+:\/\/) #PROTOCOL
      (([\w-]+\.)?) # LOWEST TLD
      ([\w-]+) # 2nd LEVEL TLD
      (\.[a-z]+) # TOP TLD
      ((:\d+)?) # PORT
      ([\S|\?]*) # PATH, QUERYSTRING AND FRAGMENT
    )$
  /ix

  text.ensure_string.strip =~ regex
end
